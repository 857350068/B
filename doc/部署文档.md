bash

编辑



# 1. 设置固定 IP (替换为您的网关，例如 192.168.1.1)
sudo vi /etc/sysconfig/network-scripts/ifcfg-ens33
# 修改以下内容：
BOOTPROTO=static
IPADDR=192.168.169.100
NETMASK=255.255.255.0
GATEWAY=192.168.169.1
DNS1=8.8.8.8

# 2. 重启网络
sudo systemctl restart network

# 3. 验证 IP
ip addr show ens33 | grep "inet"
# 应输出：inet 192.168.169.100/24
✅ 验证：
主机 ping 192.168.169.100 → 通
虚拟机 curl http://192.168.169.100 → 通

3.2 安装 Hadoop 3.3.6（伪分布式）
bash

编辑



# 1. 下载 Hadoop
wget https://archive.apache.org/dist/hadoop/common/hadoop-3.3.6/hadoop-3.3.6.tar.gz
tar -zxvf hadoop-3.3.6.tar.gz -C /opt

# 2. 配置环境变量
echo 'export HADOOP_HOME=/opt/hadoop-3.3.6' >> ~/.bashrc
echo 'export PATH=$PATH:$HADOOP_HOME/bin:$HADOOP_HOME/sbin' >> ~/.bashrc
source ~/.bashrc

# 3. 配置 core-site.xml
sudo vi $HADOOP_HOME/etc/hadoop/core-site.xml
# 添加：
<configuration>
  <property>
    <name>fs.defaultFS</name>
    <value>hdfs://localhost:9000</value>
  </property>
</configuration>

# 4. 配置 hdfs-site.xml
sudo vi $HADOOP_HOME/etc/hadoop/hdfs-site.xml
# 添加：
<configuration>
  <property>
    <name>dfs.replication</name>
    <value>1</value>
  </property>
</configuration>

# 5. 格式化 HDFS
hdfs namenode -format

# 6. 启动 Hadoop
start-dfs.sh
# 验证：jps → 应有 NameNode, DataNode
3.3 安装 Hive 3.1.2（依赖 Hadoop）
bash

编辑



# 1. 下载 Hive
wget https://archive.apache.org/dist/hive/hive-3.1.2/apache-hive-3.1.2-bin.tar.gz
tar -zxvf apache-hive-3.1.2-bin.tar.gz -C /opt

# 2. 配置环境变量
echo 'export HIVE_HOME=/opt/apache-hive-3.1.2-bin' >> ~/.bashrc
echo 'export PATH=$PATH:$HIVE_HOME/bin' >> ~/.bashrc
source ~/.bashrc

# 3. 配置 hive-site.xml
sudo vi $HIVE_HOME/conf/hive-site.xml
# 添加：
<configuration>
  <property>
    <name>hive.metastore.uris</name>
    <value>thrift://localhost:9083</value>
  </property>
  <property>
    <name>javax.jdo.option.ConnectionURL</name>
    <value>jdbc:mysql://localhost:3306/hive?createDatabaseIfNotExist=true</value>
  </property>
  <property>
    <name>javax.jdo.option.ConnectionDriverName</name>
    <value>com.mysql.jdbc.Driver</value>
  </property>
  <property>
    <name>javax.jdo.option.ConnectionUserName</name>
    <value>root</value>
  </property>
  <property>
    <name>javax.jdo.option.ConnectionPassword</name>
    <value>123456</value>
  </property>
</configuration>

# 4. 启动 Hive Metastore
hive --service metastore &

🔧 五、关键配置检查清单（避免踩坑）
表格
问题	解决方案	验证命令
虚拟机无法上网	桥接模式选择正确的物理网卡（如 Wi-Fi）	ping 8.8.8.8
Hadoop 启动失败	确保 core-site.xml 中 fs.defaultFS 为 hdfs://localhost:9000	hdfs dfs -ls /
Hive 连接 MySQL 失败	检查 hive-site.xml 中的 MySQL 密码和数据库名	mysql -u root -p123456 -e "SHOW DATABASES;"

端口被占用	检查 `netstat -tuln	grep 8080`
