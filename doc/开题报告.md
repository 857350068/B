// HiveJdbcUtil.java (核心类)
public class HiveJdbcUtil {
    public static List<Map<String, Object>> query(String sql) {
        try (Connection conn = getConnection();
             Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery(sql)) {
            
            List<Map<String, Object>> result = new ArrayList<>();
            ResultSetMetaData metaData = rs.getMetaData();
            
            while (rs.next()) {
                Map<String, Object> row = new LinkedHashMap<>();
                for (int i = 1; i <= metaData.getColumnCount(); i++) {
                    row.put(metaData.getColumnName(i), rs.getObject(i));
                }
                result.add(row);
            }
            return result;
        }
    }
}

// 使用示例：获取组织效能指数
List<Map<String, Object>> data = HiveJdbcUtil.query(
    "SELECT AVG(value) AS org_efficiency " +
    "FROM hr_analytic_data " +
    "WHERE category_id=1 AND period='MONTH'"
);

// SecurityConfig.java
@Configuration
@EnableWebSecurity
public class SecurityConfig {
    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        http
            .authorizeRequests()
                .antMatchers("/admin/**").hasRole("HR_ADMIN")  // 仅HR管理员可访问
                .antMatchers("/api/data/**").hasAnyRole("HR_ADMIN", "DEPT_HEAD", "MANAGEMENT")
                .antMatchers("/api/warning/**").hasAnyRole("HR_ADMIN", "DEPT_HEAD")
            .and()
            .addFilterBefore(jwtAuthenticationFilter(), UsernamePasswordAuthenticationFilter.class);
        return http.build();
    }
}

// DataCleanService.java (数据清洗)
@Service
public class DataCleanService {
    public List<Map<String, Object>> cleanData(List<Map<String, Object>> rawData) {
        // 1. 空值填充
        rawData.forEach(row -> {
            if (row.get("value") == null) {
                row.put("value", 0.0f);
            }
        });
        
        // 2. 异常值处理 (如离职率>100%)
        rawData.forEach(row -> {
            if ((Float) row.get("value") > 100.0f) {
                row.put("value", 100.0f);
            }
        });
        return rawData;
    }
}
