@Configuration
@EnableWebSecurity
public class SecurityConfig {

    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        http
            .csrf().disable()
            .sessionManagement().sessionCreationPolicy(SessionCreationPolicy.STATELESS)
            .and()
            .authorizeRequests()
                .antMatchers("/api/auth/**").permitAll()
                .antMatchers("/swagger-ui/**", "/v3/api-docs/**").permitAll()
                .anyRequest().authenticated()
            .and()
            .addFilterBefore(jwtAuthenticationFilter(), UsernamePasswordAuthenticationFilter.class);

        return http.build();
    }

    @Bean
    public JwtAuthenticationFilter jwtAuthenticationFilter() {
        return new JwtAuthenticationFilter();
    }
}

public class HiveJdbcUtil {
    private static final Logger logger = LoggerFactory.getLogger(HiveJdbcUtil.class);
    
    public static Connection getConnection() throws SQLException {
        String url = "jdbc:hive2://localhost:10000/hr_db";
        Properties properties = new Properties();
        properties.setProperty("user", "hive");
        properties.setProperty("password", "hive");
        return DriverManager.getConnection(url, properties);
    }

    public static List<Map<String, Object>> query(String sql) {
        try (Connection conn = getConnection();
             Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery(sql)) {
            
            List<Map<String, Object>> result = new ArrayList<>();
            ResultSetMetaData metaData = rs.getMetaData();
            int columnCount = metaData.getColumnCount();

            while (rs.next()) {
                Map<String, Object> row = new LinkedHashMap<>();
                for (int i = 1; i <= columnCount; i++) {
                    String columnName = metaData.getColumnName(i);
                    Object value = rs.getObject(i);
                    row.put(columnName, value);
                }
                result.add(row);
            }
            return result;
        } catch (SQLException e) {
            logger.error("Hive query failed: {}", sql, e);
            throw new RuntimeException("Hive query error", e);
        }
    }
}
2. 数据同步服务（DataSyncService.java）

编辑



@Service
public class DataSyncService {

    @Autowired
    private JdbcTemplate jdbcTemplate;

    @Autowired
    private HiveJdbcUtil hiveJdbcUtil;

    public void syncFromMysqlToHive() {
        // 1. 从 MySQL 获取基础数据
        List<Map<String, Object>> mysqlData = jdbcTemplate.queryForList(
            "SELECT * FROM employee_profile WHERE is_deleted = 0"
        );

        // 2. 清洗数据（示例：填充空值）
        List<Map<String, Object>> cleanedData = cleanData(mysqlData);

        // 3. 写入 Hive
        try (Connection hiveConn = HiveJdbcUtil.getConnection();
             PreparedStatement ps = hiveConn.prepareStatement(
                 "INSERT INTO TABLE hr_analytic_data VALUES (?, ?, ?, ?, ?, ?)"
             )) {
            
            for (Map<String, Object> row : cleanedData) {
                ps.setLong(1, (Long) row.get("id"));
                ps.setLong(2, (Long) row.get("category_id"));
                ps.setLong(3, (Long) row.get("dept_id"));
                ps.setFloat(4, (Float) row.get("value"));
                ps.setString(5, (String) row.get("period"));
                ps.setTimestamp(6, new Timestamp(System.currentTimeMillis()));
                ps.addBatch();
            }
            ps.executeBatch();
        } catch (SQLException e) {
            throw new RuntimeException("Hive sync failed", e);
        }
    }

    private List<Map<String, Object>> cleanData(List<Map<String, Object>> data) {
        // 空值填充示例
        for (Map<String, Object> row : data) {
            if (row.get("value") == null) {
                row.put("value", 0.0f);
            }
        }
        return data;
    }
}

编辑



@Aspect
@Component
public class LogAspect {

    private static final Logger logger = LoggerFactory.getLogger(LogAspect.class);

    @Before("@annotation(log)")
    public void before(JoinPoint joinPoint, Log log) {
        logger.info("操作开始: {} {}", joinPoint.getSignature().getDeclaringTypeName(), joinPoint.getSignature().getName());
    }

    @AfterReturning(returning = "result", pointcut = "@annotation(log)")
    public void afterReturning(JoinPoint joinPoint, Log log, Object result) {
        logger.info("操作成功: {} -> {}", joinPoint.getSignature().getName(), result);
    }

    @AfterThrowing(throwing = "e", pointcut = "@annotation(log)")
    public void afterThrowing(JoinPoint joinPoint, Log log, Exception e) {
        logger.error("操作失败: {} -> {}", joinPoint.getSignature().getName(), e.getMessage(), e);
    }
}
