package com.hr.config;

import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.security.config.annotation.web.builders.HttpSecurity;
import org.springframework.security.config.annotation.web.configuration.EnableWebSecurity;
import org.springframework.security.config.http.SessionCreationPolicy;
import org.springframework.security.web.SecurityFilterChain;
import org.springframework.security.web.authentication.UsernamePasswordAuthenticationFilter;

/**
 * RBAC 权限控制核心配置
 * 依据：开题报告 1.2 节 "数据安全" + 3.2.1 节角色权限要求
 * 
 * 功能说明：
 * 1. 仅 HR_ADMIN 可访问 /admin/** 路径
 * 2. DEPT_HEAD 仅可访问本部门数据 (通过 DataFilterServiceImpl 实现)
 * 3. 使用 JWT 无状态认证
 */
@Configuration
@EnableWebSecurity
public class SecurityConfig {

    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        http
            .csrf().disable()
            .sessionManagement().sessionCreationPolicy(SessionCreationPolicy.STATELESS)
            .and()
            .authorizeRequests()
                .antMatchers("/api/auth/**").permitAll()  // 登录接口开放
                .antMatchers("/admin/**").hasRole("HR_ADMIN")  // 仅管理员可访问
                .antMatchers("/api/data/**").hasAnyRole("HR_ADMIN", "DEPT_HEAD", "MANAGEMENT") // 多角色权限
                .antMatchers("/api/warning/**").hasAnyRole("HR_ADMIN", "DEPT_HEAD")
            .and()
            .addFilterBefore(jwtAuthenticationFilter(), UsernamePasswordAuthenticationFilter.class);
        return http.build();
    }
    
    private JwtAuthenticationFilter jwtAuthenticationFilter() {
        return new JwtAuthenticationFilter();
    }
}

package com.hr.model.entity;

import lombok.Data;
import javax.persistence.*;
import java.util.Date;

/**
 * 预警规则实体类
 * 依据：开题报告 3.2.2 节 "预警模型管理" + 1.2 节 "预测性管理"
 * 
 * 字段说明：
 * id: 主键
 * rule_type: 规则类型 (FLIGHT_RISK/TALENT_SHORTAGE)
 * threshold: 阈值 (如 8.0 表示 8%)
 * effective: 是否生效 (true=生效, false=未生效)
 * update_log: 修改日志 (JSON 格式存储)
 * created_at: 创建时间
 */
@Data
@Entity
@Table(name = "warning_rule")
public class WarningRule {

    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(name = "rule_type", length = 20)
    private String ruleType; // FLIGHT_RISK, TALENT_SHORTAGE

    @Column(name = "threshold")
    private Double threshold; // 预警阈值

    @Column(name = "effective")
    private Boolean effective; // 是否生效

    @Column(name = "update_log", length = 1000)
    private String updateLog; // JSON 格式: [{"time":"2026-02-20", "threshold":8.0}]

    @Column(name = "created_at")
    private Date createdAt;
}

package com.hr.service.impl;

import com.hr.mapper.EmployeeProfileMapper;
import com.hr.model.dto.DataVO;
import com.hr.model.util.HiveUtil;
import com.hr.model.util.DataCleanUtil;
import com.hr.service.DataSyncService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

import java.util.List;
import java.util.Map;

/**
 * 数据同步服务实现
 * 依据：开题报告 3.2.2 节 "数据管理：同步 MySQL 基础数据至 Hive"
 * 
 * 功能说明：
 * 1. 从 MySQL 查询员工基础数据
 * 2. 执行数据清洗 (空值处理、异常值修正)
 * 3. 写入 Hive 分区表 (hr_analytic_data)
 * 4. 自动添加分区 (period = '2026')
 */
@Service
@Transactional
public class DataSyncServiceImpl implements DataSyncService {

    @Autowired
    private EmployeeProfileMapper employeeProfileMapper;
    
    @Autowired
    private HiveUtil hiveUtil;

    @Override
    public void syncToHive() {
        // 1. 从 MySQL 查询基础数据
        List<Map<String, Object>> mysqlData = employeeProfileMapper.selectAll();
        
        // 2. 数据清洗 (开题报告 3.3 节)
        List<Map<String, Object>> cleanedData = DataCleanUtil.cleanData(mysqlData);
        
        // 3. 写入 Hive (开题报告 4.2.5 节)
        hiveUtil.batchInsert("hr_analytic_data", cleanedData, "2026"); // 分区: 2026
    }
}

💡 为什么这样配置？
8080 端口是 SpringBoot 默认端口（符合毕业设计规范）
allowed-origins 包含开发环境和生产环境（答辩时可解释）
allow-credentials: true 允许携带Cookie（实现登录状态保持）

3. Nginx 配置（deploy/nginx-config/hr-datacenter.conf）
nginx

编辑



server {
    listen 80;
    server_name your-domain.com;  # 替换为实际域名

    # 前端静态资源路径
    location /hr-datacenter/ {
        root /usr/share/nginx/html;
        index index.html;
        try_files $uri $uri/ /hr-datacenter/index.html;
    }

    # 后端API代理
    location /api/ {
        proxy_pass http://localhost:8080;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # 静态资源缓存
    location /static/ {
        expires 30d;
        add_header Cache-Control "public";
    }
}
✅ 关键设计：
/hr-datacenter/ 作为前端根路径（解决子路径问题）
/api/ 代理到后端8080端口（解决跨域）
try_files 确保Vue路由正常工作

📝 五、毕业设计答辩关键话术
1. 为什么选择前后端分离？
"根据开题报告1.2节'技术架构革新'要求，我们采用前后端分离架构：
后端使用SpringBoot提供RESTful API（符合开题报告4.2节技术要求）
前端使用Vue构建响应式界面（匹配开题报告3.2.3节'前端界面功能'）
这种架构使系统更易维护、扩展，且符合企业级应用标准。"
2. 如何解决跨域问题？
"我们通过Nginx代理解决跨域：
前端部署在 /hr-datacenter/ 路径
Nginx将 /api/ 请求代理到后端8080端口
这样既避免了浏览器跨域限制，又符合开题报告3.3节'数据安全'要求。"
3. 为什么设置 VUE_APP_BASE_URL？
"为适应企业级系统多级部署场景，我们在前端设置了子路径：
VUE_APP_BASE_URL="/hr-datacenter/"
这样系统可以部署在https://your-domain.com/hr-datacenter/，而无需修改前端代码。"
